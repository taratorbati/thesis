    def _surface_runoff(self, rain):
        t = self.theta
        if rain > t['theta3']:
            phi2 = (rain - t['theta3'])**2 / (rain + 4 * t['theta3'])
        else:
            phi2 = 0.0
        return np.full(self.N, phi2)

    def _redistribute_runoff(self, phi2, rain):
        """
        Redistribute surface runoff to lower neighbors.
        Returns phi2_in array (water received from uphill agents).
        """
        if self.runoff_mode == 'none':
            return np.zeros(self.N)

        elif self.runoff_mode == 'simple':
            # Approach A: send phi2 to lower neighbors, instant infiltration
            phi2_in = np.zeros(self.N)
            for n in range(self.N):
                if phi2[n] > 0 and self.Nr[n] > 0:
                    per_neighbor = phi2[n] / self.Nr[n]
                    for m in self.sends_to[n]:
                        phi2_in[m] += per_neighbor
            return phi2_in

        elif self.runoff_mode == 'cascade':
            # Approach B: process top-to-bottom, re-run SCS at each level
            # Each agent's phi2 flows to lower neighbors as additional water.
            # Lower agents re-run SCS on (received runoff) to determine
            # how much infiltrates vs runs off further.
            t = self.theta
            phi2_in = np.zeros(self.N)
            pending = np.zeros(self.N)  # runoff arriving from uphill

            # Seed: each agent's own phi2 is sent to its lower neighbors
            for n in self.sorted_agents:
                # Total runoff leaving this agent: own phi2 + re-runoff of received water
                to_send = phi2[n] + 0.0  # start with own runoff

                # Add received water from uphill
                if pending[n] > 0:
                    # Re-run SCS on received water
                    effective = pending[n]
                    if effective > t['theta3']:
                        re_runoff = (effective - t['theta3'])**2 / \
                                    (effective + 4 * t['theta3'])
                    else:
                        re_runoff = 0.0

                    # What infiltrates
                    phi2_in[n] = effective - re_runoff
                    # What runs off further
                    to_send += re_runoff

                # Send total runoff to lower neighbors
                if to_send > 0 and self.Nr[n] > 0:
                    per_neighbor = to_send / self.Nr[n]
                    for m in self.sends_to[n]:
                        pending[m] += per_neighbor
                elif to_send > 0 and self.Nr[n] == 0:
                    # Sink agent: all pending water infiltrates (pools)
                    phi2_in[n] += pending[n]

            return phi2_in

        else:
            raise ValueError(f"Unknown runoff_mode: {self.runoff_mode}")

    def _deep_drainage(self, rain, phi2):
        t = self.theta
        fc_total = t['theta6'] * t['theta5']
        excess = self.x1 + rain - phi2 - fc_total
        return np.where(excess > 0, t['theta4'] * excess, 0.0)

    def _water_exchange(self):
        t = self.theta
        fc_total = t['theta6'] * t['theta5']
        phi4_in = np.zeros(self.N)
        phi4_out = np.zeros(self.N)
        for n in range(self.N):
            excess = self.x1[n] - fc_total
            if excess > 0 and self.Nr[n] > 0:
                outflow_per_neighbor = excess / self.Nr[n]
                phi4_out[n] += excess
                for m in self.sends_to[n]:
                    phi4_in[m] += outflow_per_neighbor
        return phi4_in, phi4_out
